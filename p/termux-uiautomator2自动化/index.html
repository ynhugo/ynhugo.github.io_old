import uiautomator2 as u2

def main():
    u = u2.connect('192.168.3.193')    
    
    # 输入设备
    # print(u.info)

    # QQ好友点赞
    u.app_start('com.tencent.mobileqq')    
    u.click(88,173)
    u(text='夜闹').click()
    u.click(975,695)
    for i in range(10):
      u.click(1018,1157)

if __name__ == '__main__':
    main()

import uiautomator2 as u2

def main():
    u = u2.connect('192.168.3.193')    
    
    # 元素定位
    xml = u.dump_hierarchy()
    # print(xml)
    # 保存到文件（方便分析）
    with open("ui_dump.xml", "w", encoding="utf-8") as f:
      f.write(xml)

if __name__ == '__main__':
    main()
